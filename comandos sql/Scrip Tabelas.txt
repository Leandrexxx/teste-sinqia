USE TesteSinqia;
GO

CREATE TABLE CIDADES
(
CODIGO_CIDADE INT IDENTITY NOT NULL,
NOME VARCHAR(100) NOT NULL,
ESTADO CHAR(2) NOT NULL,
CEP_INICIAL VARCHAR(8) NOT NULL,
CEP_FINAL VARCHAR(8) NOT NULL,
CONSTRAINT PK_CIDADES_CODIGO PRIMARY KEY (CODIGO_CIDADE)
);
GO

CREATE TABLE CLIENTES
(
CODIGO_CLIENTE INT IDENTITY NOT NULL,
CGC_CPF_CLIENTE VARCHAR(14) NOT NULL,
NOME VARCHAR(100) NOT NULL,
TELEFONE VARCHAR(14) NULL,
ENDERECO VARCHAR(100) NULL,
BAIRRO VARCHAR(50) NULL,
COMPLEMENTO VARCHAR(50) NULL,
EMAIL VARCHAR(100) NULL,
CODIGO_CIDADE INT,
CEP VARCHAR(8) NULL,
CONSTRAINT PK_CLIENTES_CODIGO PRIMARY KEY (CODIGO_CLIENTE),
CONSTRAINT FK_CLIENTE_CIDADES FOREIGN KEY (CODIGO_CIDADE)
REFERENCES CIDADES(CODIGO_CIDADE)
);
GO
